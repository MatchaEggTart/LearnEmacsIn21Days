* [[https://github.com/slime/slime][slime]]

** 配置

#+begin_src emacs-lisp
  (use-package slime
    :config
    (progn
      (setq inferior-lisp-program "sbcl")
      (setq tab-width 2)
      (slime-setup '(slime-repl slime-fancy slime-banner)
		   ;; (setq slime-contribs '(slime-fancy slime-cl-indent))
		   )))

  (provide 'init-lisp)
#+end_src
